* API Reference

** Hyperdream.Router

*** Types

#+BEGIN_SRC ocaml
type meth = [ `GET | `POST | `PUT | `PATCH | `DELETE ]

type request = {
  meth : H1.Method.t;
  path : string;
  query : string option;
  headers : H1.Headers.t;
  body : string;
  reqd : H1.Reqd.t;
}

type response =
  | Html of { status : H1.Status.t; headers : (string * string) list; body : string }
  | Streaming
  | No_content
  | Not_found

type handler = request -> response Lwt.t
#+END_SRC

*** Functions

- =add : (meth * string) -> handler -> unit= — Register a route
- =find : meth -> string -> handler option= — Look up a handler
- =dispatch : H1.Reqd.t -> unit Lwt.t= — Dispatch a request through the router
- =parse_target : string -> string * string option= — Split target into path and query
- =make_request : H1.Reqd.t -> request Lwt.t= — Parse an H1 request into a framework request
- =respond_string : H1.Reqd.t -> ?headers:(string * string) list -> status:H1.Status.t -> body:string -> unit -> unit= — Send a string response
